```typescript
/**

 *   manifestUrl: '../versions.json'  (relative to a deployed page at /<versionSlot>/...)

 *

export async function injectWidgetForVersion(
  workdir: string,
  versionSlot: string,
  repoMeta: RepoMeta,
): Promise<number>;
```

**Step 1 — extend `src/branch-manager.ts`:**

Add the new helper at the bottom of the file, after `writeIndexHtml`:

```typescript
import { injectWidgetIntoHtmlFiles } from './widget-injector.js';

// [LAW:single-enforcer] All writes to the gh-pages worktree live in this module.
// The widget script is generated by the pure helper in widget-injector.ts; this
// function is the sole I/O enforcer that lands the script tag in deployed HTML files.
// [LAW:dataflow-not-control-flow] Runs unconditionally on every deploy. Empty html
// list returns 0 from the underlying walker — no guarded skip. The relative URLs
// are derived purely from versionSlot (data-driven, [LAW:one-source-of-truth]).
export async function injectWidgetForVersion(
  workdir: string,
  versionSlot: string,
  _repoMeta: RepoMeta,
): Promise<number> {
  const versionDir = path.join(workdir, versionSlot);
  return injectWidgetIntoHtmlFiles(versionDir, {
    manifestUrl: '../versions.json',
    indexUrl: '../',
    currentVersion: versionSlot,
  });
}
```

Note: `_repoMeta` is accepted for future-proofing (Plan 5 PR previews may need owner/repo for absolute URL generation), but unused in v1. Prefix with underscore to silence the lint rule. <threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| placeContent output → injectWidgetForVersion input | The base-path-corrected HTML files are the input to widget injection. If injection ran BEFORE placeContent, the base-path rewriter could corrupt the injected script tag (e.g., rewrite `'../versions.json'` to a different path). Wiring order is part of the threat surface. |
| sibling version directories on the gh-pages branch | Previous deployments share the worktree. The injector must only touch the current version slot (D-03), never sibling directories. |
| dist/index.js bundle | The published Action runs from this file. If the bundle is stale, the deployed Action will not include the widget even though source tests pass. |
