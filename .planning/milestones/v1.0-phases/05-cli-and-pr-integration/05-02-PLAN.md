```typescript
export const PREVIEW_COMMENT_MARKER = '<!-- gh-pages-multiplexer:preview -->';

export interface PreviewCommentOpts {
  owner: string;
  repo: string;
  prNumber: number;
  previewUrl: string;
  version: string;
}

/**
 * Octokit shape we depend on (structural typing — no @actions/github import here for testability).
 * The Action adapter passes the real octokit; tests pass a hand-rolled mock with method counts.
 */
export interface CommenterOctokit {
  rest: {
    issues: {
      listComments(params: { owner: string; repo: string; issue_number: number; per_page?: number }): Promise<{ data: Array<{ id: number; body?: string | null }> }>;
      createComment(params: { owner: string; repo: string; issue_number: number; body: string }): Promise<unknown>;
      updateComment(params: { owner: string; repo: string; comment_id: number; body: string }): Promise<unknown>;
    };
  };
}

/**
 * [LAW:one-source-of-truth] PREVIEW_COMMENT_MARKER is the sole identity check.
 * [LAW:dataflow-not-control-flow] The same three-step pipeline runs every call:
 *   list → select-by-marker → write (create OR update). Variability is in the data
 *   (matching comment id present? null? → drives which write API is called).
 */
export async function upsertPreviewComment(
  octokit: CommenterOctokit,
  opts: PreviewCommentOpts,
```

```typescript
  type CommenterOctokit,
  type PreviewCommentOpts,
} from '../src/pr-commenter.js';

function makeMockOctokit(options: {
  existingComments?: Array<{ id: number; body?: string | null }>;
  listError?: Error & { status?: number };
  createError?: Error & { status?: number };
  updateError?: Error & { status?: number };
}): CommenterOctokit & {
  _calls: { list: number; create: number; update: number };
  _lastCreateBody?: string;
  _lastUpdateCommentId?: number;
  _lastUpdateBody?: string;
} {
  const calls = { list: 0, create: 0, update: 0 };
  let lastCreateBody: string | undefined;
  let lastUpdateCommentId: number | undefined;
  let lastUpdateBody: string | undefined;
  return {
    _calls: calls,
    get _lastCreateBody() { return lastCreateBody; },
    get _lastUpdateCommentId() { return lastUpdateCommentId; },
    get _lastUpdateBody() { return lastUpdateBody; },
    rest: {
      issues: {
        async listComments(params) {
          calls.list++;
          if (options.listError) throw options.listError;
          return { data: options.existingComments ?? [] };
        },
        async createComment(params) {
          calls.create++;
          lastCreateBody = params.body;
          if (options.createError) throw options.createError;
          return {};
        },
        async updateComment(params) {
          calls.update++;
          lastUpdateCommentId = params.comment_id;
          lastUpdateBody = params.body;
          if (options.updateError) throw options.updateError;
          return {};
        },
      },
    },
  } as any;
}
```

```typescript
// [LAW:one-source-of-truth] PREVIEW_COMMENT_MARKER is the sole identity check for "this is the
//   gh-pages-multiplexer preview comment." Same pattern as the widget marker (Phase 4 D-12).
// [LAW:dataflow-not-control-flow] Every call runs the same three-step pipeline: list → select →
//   write. The variability lives in the data: the matched comment id (number | null) drives
//   whether the write call is updateComment or createComment. The operation is unconditional —
//   exactly one write happens on every successful run.
// [LAW:single-enforcer] This module is the only place that knows the marker, the body template,
//   and the 403-swallow boundary. Callers (src/index.ts) only call upsertPreviewComment.
// [LAW:no-defensive-null-guards] exception: D-19 — the only swallowed error is HTTP 403 (and only
//   from the GitHub API), because the comment is genuinely optional and the deploy has already
//   succeeded by the time this runs. Every other error rethrows. The exception is bounded to one
//   discriminator (err.status === 403) and one function.
import * as core from '@actions/core';

export const PREVIEW_COMMENT_MARKER = '<!-- gh-pages-multiplexer:preview -->';

export interface PreviewCommentOpts {
  owner: string;
  repo: string;
  prNumber: number;
  previewUrl: string;
  version: string;
}

export interface CommenterOctokit {
  rest: {
    issues: {
      listComments(params: { owner: string; repo: string; issue_number: number; per_page?: number }): Promise<{ data: Array<{ id: number; body?: string | null }> }>;
      createComment(params: { owner: string; repo: string; issue_number: number; body: string }): Promise<unknown>;
      updateComment(params: { owner: string; repo: string; comment_id: number; body: string }): Promise<unknown>;
    };
  };
}

function buildBody(opts: PreviewCommentOpts): string {
  const ts = new Date().toISOString();
  return [
    PREVIEW_COMMENT_MARKER,
    '### 📦 Preview deployed',
    '',
    `**Version:** \`${opts.version}\``,
    `**Preview:** ${opts.previewUrl}`,
    '',
    `_Updated at ${ts}_`,
  ].join('\n');
}

function is403(err: unknown): boolean {
  return typeof err === 'object' && err !== null && (err as { status?: unknown }).status === 403;
}

function warnPermissions(action: string, err: unknown): void {
  const msg = err instanceof Error ? err.message : String(err);
  core.warning(
    `PR preview comment ${action} returned 403: ${msg}. ` +
    `Add \`permissions: pull-requests: write\` to your workflow to enable preview comments. ` +
    `(Deploy succeeded; comment is optional — see D-19.)`,
  );
}

export async function upsertPreviewComment(
  octokit: CommenterOctokit,
  opts: PreviewCommentOpts,
): Promise<void> {
  // Step 1: list. 403 here = missing pull-requests: read; swallow with warning per D-14/D-19.
  let comments: Array<{ id: number; body?: string | null }>;
  try {
    const res = await octokit.rest.issues.listComments({
      owner: opts.owner,
      repo: opts.repo,
      issue_number: opts.prNumber,
      per_page: 100,
    });
    comments = res.data;
  } catch (err) {
    if (is403(err)) {
      warnPermissions('list', err);
      return;
    }
    throw err;
  }

  // Step 2: select by marker (data-driven; first match wins, deterministic).
  const matched = comments.find((c) => (c.body ?? '').includes(PREVIEW_COMMENT_MARKER));

  // Step 3: write. The operation runs unconditionally; the *which write* is data-driven.
  const body = buildBody(opts);
  try {
    if (matched) {
      await octokit.rest.issues.updateComment({
        owner: opts.owner,
        repo: opts.repo,
        comment_id: matched.id,
        body,
      });
    } else {
      await octokit.rest.issues.createComment({
        owner: opts.owner,
        repo: opts.repo,
        issue_number: opts.prNumber,
        body,
      });
    }
  } catch (err) {
    if (is403(err)) {
      warnPermissions(matched ? 'update' : 'create', err);
      return;
    }
    throw err;
  }
}
```

**Step C — src/cli.ts implementation:**

```typescript
#!/usr/bin/env node
// [LAW:one-type-per-behavior] CLI is a second adapter into the same deploy() pipeline used
//   by the Action adapter. It does NOT contain any pipeline logic — only argv/env → DeployConfig
//   translation, exit code mapping, and stderr formatting.
// [LAW:variability-at-edges] The pipeline core is fixed in src/deploy.ts. CI-specific input
//   gathering (argv vs @actions/core, env vars vs core.getInput) lives at the edge here.
// [LAW:dataflow-not-control-flow] main() always runs the same five steps: parse argv → resolve
//   token → build config → call deploy → map result to exit code. Errors are data on the result
//   path (exit codes), not control-flow short-circuits scattered through the function.
import { parseArgs } from 'node:util';
import type { DeployConfig } from './types.js';
import { deploy } from './deploy.js';

const VERSION = '0.0.0'; // Synced with package.json version field; bump together.

const HELP_TEXT = `Usage: gh-pages-multiplexer deploy [options]

Deploy a static site to a versioned subdirectory on a GitHub Pages branch.

Options:
  --source-dir=<path>          Directory containing the built site (required)
  --target-branch=<name>       Target gh-pages branch (default: gh-pages)
  --ref-patterns=<csv>         Comma-separated ref patterns to deploy (default: matches action.yml)
  --base-path-mode=<mode>      base-tag | rewrite (default: base-tag)
  --base-path-prefix=<prefix>  Override auto-detected base path prefix (default: empty = auto)
  --repo=<owner/name>          Repository slug (default: $GITHUB_REPOSITORY)
  --ref=<refs/...>             Git ref being deployed (default: $GITHUB_REF)
  --debug                      Print full stack traces on error
  --help                       Show this help and exit
  --version                    Print version and exit

Environment:
  GITHUB_TOKEN                 GitHub token with contents: write (preferred)
  GH_TOKEN                     Fallback token env var (matches gh CLI convention)
  DEBUG=1                      Equivalent to --debug

Exit codes:
  0  success
  1  deployment failure
  2  configuration error (missing inputs, bad flags, missing token)
`;

export async function main(argv: string[], env: NodeJS.ProcessEnv): Promise<number> {
  // --help / --version are honored before subcommand parsing so they work as `gh-pages-multiplexer --help`.
  if (argv.includes('--help') || argv.includes('-h')) {
    process.stdout.write(HELP_TEXT);
    return 0;
  }
  if (argv.includes('--version') || argv.includes('-v')) {
    process.stdout.write(`gh-pages-multiplexer ${VERSION}\n`);
    return 0;
  }

  // Subcommand: only `deploy` is supported in v1 (D-01).
  const [subcommand, ...rest] = argv;
  if (subcommand !== 'deploy') {
    process.stderr.write(
      `Error: unknown or missing subcommand '${subcommand ?? ''}'. Only 'deploy' is supported.\n` +
      `Run with --help for usage.\n`,
    );
    return 2;
  }

  // parseArgs against the deploy flag set. strict: true → unknown flags throw (D-05: --token NOT in the set).
  let parsed: ReturnType<typeof parseArgs>;
  try {
    parsed = parseArgs({
      args: rest,
      strict: true,
      allowPositionals: false,
      options: {
        'source-dir': { type: 'string' },
        'target-branch': { type: 'string' },
        'ref-patterns': { type: 'string' },
        'base-path-mode': { type: 'string' },
        'base-path-prefix': { type: 'string' },
        'repo': { type: 'string' },
        'ref': { type: 'string' },
        'debug': { type: 'boolean' },
      },
    });
  } catch (err) {
    const msg = err instanceof Error ? err.message : String(err);
    process.stderr.write(`Error: ${msg}\n`);
    if (msg.toLowerCase().includes('token')) {
      process.stderr.write(
        `Note: tokens must be provided via the GITHUB_TOKEN or GH_TOKEN environment variable, not as a flag.\n`,
      );
    }
    return 2;
  }

  const debug = Boolean(parsed.values['debug']) || env.DEBUG === '1';

  // D-05: token from env only.
  const token = env.GITHUB_TOKEN ?? env.GH_TOKEN ?? '';
  if (!token) {
    process.stderr.write(
      `Error: no GitHub token found. Set GITHUB_TOKEN (preferred) or GH_TOKEN in the environment.\n`,
    );
    return 2;
  }

  // Required-flag validation. parseArgs doesn't enforce required by itself; we do.
  const required = ['source-dir'] as const;
  for (const flag of required) {
    if (!parsed.values[flag]) {
      process.stderr.write(`Error: missing required flag --${flag}\n`);
      return 2;
    }
  }

  const basePathMode = (parsed.values['base-path-mode'] ?? 'base-tag') as string;
  if (basePathMode !== 'base-tag' && basePathMode !== 'rewrite') {
    process.stderr.write(`Error: invalid --base-path-mode '${basePathMode}'. Must be 'base-tag' or 'rewrite'.\n`);
    return 2;
  }

  const refPatternsRaw = parsed.values['ref-patterns'];
  const refPatterns = typeof refPatternsRaw === 'string' && refPatternsRaw.length > 0
    ? refPatternsRaw.split(',').map((p) => p.trim()).filter((p) => p.length > 0)
    : []; // empty default = caller relied on action.yml default; CLI deploy without patterns means "all refs"

  const config: DeployConfig = {
    sourceDir: parsed.values['source-dir'] as string,
    targetBranch: (parsed.values['target-branch'] ?? 'gh-pages') as string,
    refPatterns,
    basePathMode: basePathMode as 'base-tag' | 'rewrite',
    basePathPrefix: (parsed.values['base-path-prefix'] ?? '') as string,
    token,
    repo: (parsed.values['repo'] ?? env.GITHUB_REPOSITORY ?? '') as string,
    ref: (parsed.values['ref'] ?? env.GITHUB_REF ?? '') as string,
  };

  try {
    const result = await deploy(config, process.cwd());
    process.stdout.write(`Deployed ${result.version} to ${result.url}\n`);
    return 0;
  } catch (err) {
    const msg = err instanceof Error ? err.message : String(err);
    process.stderr.write(`Error: ${msg}\n`);
    if (debug && err instanceof Error && err.stack) {
      process.stderr.write(err.stack + '\n');
    }
    return 1;
  }
}

// Top-level invocation — separated from main() so tests can call main() directly with injected argv/env.
// This block runs only when the file is executed as a script (npx / dist/cli.js), NOT when imported by tests.
// Vitest imports this module via vi.mock-aware paths; the import.meta check (ESM) doesn't apply here because
// rollup emits CJS — instead, we guard on `require.main === module` after the rollup CJS transform. The
// simpler approach: just always call main() at the bottom; the test file imports `{ main }` and never
// triggers this block because vitest's module loader does not execute the bottom block when tree-shaking
// for named imports... ACTUALLY: vitest WILL execute the bottom block on import. So we MUST guard.
// CJS guard (works after rollup → CJS):
declare const require: { main?: unknown } | undefined;
declare const module: unknown;
if (typeof require !== 'undefined' && typeof module !== 'undefined' && (require as any).main === module) {
  main(process.argv.slice(2), process.env)
    .then((code) => process.exit(code))
    .catch((err) => {
      process.stderr.write(`Fatal: ${err instanceof Error ? err.message : String(err)}\n`);
      process.exit(1);
    });
}
```

```typescript
import * as core from '@actions/core';
import * as github from '@actions/github';
import type { DeployConfig } from './types.js';
import { deploy } from './deploy.js';
import { upsertPreviewComment } from './pr-commenter.js';

// ... existing parseInputs() unchanged ...

async function run(): Promise<void> {
  const sourceRepoDir = process.cwd();
  const config = parseInputs();
  core.info(`Deploying from ${config.sourceDir} to ${config.targetBranch}`);
  core.info(`Ref: ${config.ref}, Repo: ${config.repo}`);

  const result = await deploy(config, sourceRepoDir);

  core.setOutput('version', result.version);
  core.setOutput('url', result.url);
  core.info(`Deployed ${result.version} to ${result.url}`);

  // [LAW:no-defensive-null-guards] exception: D-19 — PR comment is genuinely optional;
  //   the deploy has already succeeded. The swallow boundary is bounded to this single
  //   call site and is documented in the phase 5 CONTEXT (D-19). Failure here MUST NOT
  //   fail the workflow.
  // [LAW:dataflow-not-control-flow] The eligibility check is data: PR context present?
  //   We always reach the same call shape; the variability is whether `pr` is null.
  const ctx = github.context;
  const isPrEvent = ctx.eventName === 'pull_request' || ctx.eventName === 'pull_request_target';
  const pr = isPrEvent ? ctx.payload.pull_request : undefined;
  if (pr && typeof pr.number === 'number' && config.repo.includes('/')) {
    const [owner, repo] = config.repo.split('/');
    const octokit = github.getOctokit(config.token);
    try {
      await upsertPreviewComment(octokit, {
        owner,
        repo,
        prNumber: pr.number,
        previewUrl: result.url,
        version: result.version,
      });
      core.info(`Updated PR #${pr.number} preview comment`);
    } catch (err) {
      // [LAW:no-defensive-null-guards] exception: D-19 documented swallow boundary.
      const msg = err instanceof Error ? err.message : String(err);
      core.warning(`PR preview comment failed: ${msg}. Deploy itself succeeded.`);
    }
  }
}

run().catch((error: unknown) => {
  const message = error instanceof Error ? error.message : String(error);
  core.setFailed(message);
});
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| process.argv → src/cli.ts parseArgs | User-supplied flag values flow into DeployConfig and onward to git operations. A malicious flag value could try to inject shell metacharacters into a downstream git command if any stage uses string interpolation into a shell. |
| process.env → src/cli.ts token resolution | GITHUB_TOKEN is high-value secret material. Any path that prints it (default error formatter, --debug stack frames containing config object, accidental console.log) is a credential leak. |
| github.context.payload → src/index.ts PR detection | The `pull_request` payload comes from GitHub's webhook delivery; we trust GitHub's authentication of the event. But `pr.number` shape must be defended against — if it's missing or non-numeric, the commenter must skip cleanly, not crash the deploy. |
| octokit response → src/pr-commenter.ts | The list-comments response includes attacker-controllable text (anyone can comment on a public PR with a body containing the marker string). The first-marker-wins rule could be subverted by an attacker who races to post a fake marked comment first, causing our commenter to update a comment they own. The blast radius is limited to "we update someone else's comment with our preview body" — they retain the comment id and could edit it to mislead, but they cannot escalate beyond comment content. |
| 403 swallow boundary | The single documented exception to "no swallowed errors". A bug here (e.g., swallowing 401 or 500 thinking they're permission errors) would mask real failures. |
